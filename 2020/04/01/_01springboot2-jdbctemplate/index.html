通过JdbcTemplate实现SpringContactService中定义的数据访问操作

@Service
public class SpringContactServiceImpl implements SpringContactService {

	@Autowired
	private JdbcTemplate jdbcTemplate;

	@Override
	public void create(SpringContact entity) {
		this.jdbcTemplate.update("insert into spring_contact " 
				+ "(user_name,mobile,tel,fax)" + " values (?,?,?,?)",
				entity.getUserName(), 
				entity.getMobile(), 
				entity.getTel(), 
				entity.getFax());
	}

	@Override
	public void update(SpringContact entity) {
		this.jdbcTemplate.update("update spring_contact set mobile=?,tel=?,fax=? where id=?", 
				entity.getMobile(),
				entity.getTel(), 
				entity.getFax(), 
				entity.getId());
	}



	@Override
	public SpringContact getByUserName(String userName) {
		return this.jdbcTemplate.queryForObject("select * from spring_contact where user_name=?", 
		(resultSet, rowNum) -> {
			SpringContact entity = new SpringContact();
			entity.setId(resultSet.getInt("id"));
			entity.setUserName(resultSet.getString("user_name"));
			entity.setMobile(resultSet.getString("mobile"));
			entity.setTel(resultSet.getString("tel"));
			entity.setFax(resultSet.getString("fax"));
			return entity;
		}, userName);
	}

	@Override
	public void deleteByUserName(String userName){
		this.jdbcTemplate.update("delete from spring_contact where user_name=?",userName);
	}

	@Override
	public Integer getTotalNumbers() {
		return this.jdbcTemplate.queryForObject("select count(*) from spring_contact", Integer.class) ;
	}

}

创建对SpringContactService的单元测试用例

@RunWith(SpringRunner.class)
@SpringBootTest
class SpringContactServiceTest {

	@Autowired
	private SpringContactService springContactService;

	@Test
	void test() {

		for (int i = 0; i < 5; i++) {
			SpringContact entity = new SpringContact();
			entity.setUserName("SpringSongs"+i);
			entity.setMobile("12345678911");
			entity.setTel("0777-955888");
			entity.setFax("0777-955888");
			this.springContactService.create(entity);
		}

		Integer counts = this.springContactService.getTotalNumbers();
		Assert.assertEquals(new Integer(5), counts);
		
		
		SpringContact springContact = this.springContactService.getByUserName("SpringSongs1");
		Assert.assertEquals("SpringSongs1", springContact.getUserName());

		
		this.springContactService.deleteByUserName("SpringSongs1");
		counts = this.springContactService.getTotalNumbers();
		Assert.assertEquals(new Integer(4), counts);
		
	}

}
