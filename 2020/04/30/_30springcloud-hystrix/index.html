创建Controller

@RestController
@RequestMapping(value = "/SpringContact")
public class SpringContactController {

	@Autowired
	private SpringContactService springContactService;

	@Autowired
	private LoadBalancerClient loadBalancerClient;

	@GetMapping("/get/{id}")
	public SpringContact get(@PathVariable Long id) {
		return this.springContactService.get(id);
	}

	@GetMapping("/logSpringContactService")
	public void logSpringContactService() {
		ServiceInstance serviceInstance = this.loadBalancerClient.choose("MICROSERVICE-SPRINGCONTACT");
		System.out.print("微服务：" + serviceInstance.getInstanceId() + ":" + serviceInstance.getHost() + ":"
				+ serviceInstance.getPort());
	}
}
